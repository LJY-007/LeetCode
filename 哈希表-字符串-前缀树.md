    void insert(string word) {

        for (char c : word) {
            if (!curr->nexts.count(c)) curr->nexts[c] = new TrieNode();
            curr = curr->nexts[c];
            curr->path++;
        }
        curr->end++; 
    }
    
    bool search(string word) {
        TrieNode *curr = root;
        for (char c : word) {
            if (!curr->nexts.count(c)) return false;
            curr = curr->nexts[c];
        }
        return curr->end;
    }

        TrieNode *curr = root;
        for (char c : prefix) {
            if (!curr->nexts.count(c)) return false;
            curr = curr->nexts[c];
        }
        return true;
    }

private:
    TrieNode *root;
};
